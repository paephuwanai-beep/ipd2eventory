function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; // ชีตแรกคือข้อมูลเวร var data = sheet.getDataRange().getDisplayValues(); // ดึงข้อมูลการตั้งค่าชื่อ var settingSheet = ss.getSheetByName("Settings"); var settingsData = {}; if (settingSheet) { var sData = settingSheet.getDataRange().getDisplayValues(); if (sData.length > 1) { for (var i = 0; i < sData[0].length; i++) { settingsData[sData[0][i]] = sData[1][i]; } } } return ContentService.createTextOutput(JSON.stringify({status: "success", data: data, settings: settingsData})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var lock = LockService.getScriptLock(); lock.tryLock(10000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var action = data.action; // ------------------------------------------------------------- // ระบบบันทึกการตั้งค่าชื่อ (บันทึกลงชีตใหม่ชื่อ Settings อัตโนมัติ) // ------------------------------------------------------------- if (action === "save_settings") { var settingSheet = ss.getSheetByName("Settings"); if (!settingSheet) { settingSheet = ss.insertSheet("Settings"); } settingSheet.clear(); var keys = Object.keys(data.settings); var values = keys.map(function(k) { return data.settings[k]; }); settingSheet.appendRow(keys); settingSheet.appendRow(values); return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "อัปเดตการตั้งค่าสำเร็จ"})) .setMimeType(ContentService.MimeType.JSON); } // ------------------------------------------------------------- // ระบบบันทึกข้อมูลเวร (ระบบเดิม) // ------------------------------------------------------------- var sheet = ss.getSheets()[0]; var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(function(h) { return String(h).trim(); }); var uidColIndex = headers.indexOf("UID"); var rowIndex = -1; if ((action === "update" || action === "delete") && uidColIndex > -1) { var lastRow = sheet.getLastRow(); if (lastRow > 1) { var searchData = sheet.getRange(2, uidColIndex + 1, lastRow - 1, 1).getValues(); for (var j = 0; j < searchData.length; j++) { if (String(searchData[j][0]).trim() === String(data.uid).trim()) { rowIndex = j + 2; break; } } } } if (action === "delete") { if (rowIndex > -1) { sheet.deleteRow(rowIndex); return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "ลบข้อมูลสำเร็จ"})) .setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": "ไม่พบ UID ดังกล่าว"})) .setMimeType(ContentService.MimeType.JSON); } } var obj = data.datasetObj; var oldRow = []; if (rowIndex > -1) { oldRow = sheet.getRange(rowIndex, 1, 1, headers.length).getValues()[0]; } var newRow = []; for (var i = 0; i < headers.length; i++) { var headerName = headers[i]; if (headerName === "UID") { newRow.push(data.uid); } else if (headerName === "วันที่") { newRow.push(data.date); } else if (headerName === "เวร") { newRow.push(data.shift); } else if (obj && obj[headerName] !== undefined) { newRow.push(obj[headerName]); } else { if (rowIndex > -1 && oldRow[i] !== undefined) { newRow.push(oldRow[i]); } else { newRow.push(""); } } } if (rowIndex > -1 && action === "update") { sheet.getRange(rowIndex, 1, 1, newRow.length).setValues([newRow]); } else { sheet.appendRow(newRow); } return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})) .setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } }
